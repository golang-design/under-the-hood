{{/* Header language switcher. Design goals: * Works TODAY with a single language (zh-cn). The trigger shows the current language and the dropdown lists whatever languages exist. * Requires NO further template edits when English/German content is added: real translations come from Hugo's site languages / page translations, so they light up automatically. Until a planned language has content it is shown as a disabled "coming soon" entry purely for discoverability. How the entries are decided: 1. $real - every language Hugo actually knows about. We union the site's configured languages (.Site.Languages, so the home page always has an entry) with the current page's translations (.Translations + .AllTranslations) so per-page links are correct when present. 2. $planned - the roadmap languages (en, de). Any planned language that is already "real" is dropped so we never show a duplicate or a dead link once it ships. */}} {{/* --- collect real languages, keyed by lang code, value is best link target --- */}} {{ $real := dict }} {{ range .Site.Languages }} {{ $real = merge $real (dict .Lang (dict "name" .LanguageName "lang" .Lang "url" "" )) }} {{ end }} {{ range .AllTranslations }} {{ $real = merge $real (dict .Language.Lang (dict "name" .Language.LanguageName "lang" .Language.Lang "url" .Permalink )) }} {{ end }} {{ range .Translations }} {{ $real = merge $real (dict .Language.Lang (dict "name" .Language.LanguageName "lang" .Language.Lang "url" .Permalink )) }} {{ end }} {{/* --- planned roadmap languages; pruned of anything already real --- */}} {{ $planned := slice (dict "lang" "en" "name" "English") (dict "lang" "de" "name" "Deutsch") }} {{ $curLang := .Site.Language.Lang }} {{ $curName := .Site.Language.LanguageName }}
    {{/* real languages first, sorted by code for a stable order */}} {{ range $code, $l := $real }}
  • {{ if eq $code $curLang }} {{ $l.name }} {{ else if $l.url }} {{/* ?lang persists the explicit choice (see partial lang-redirect) so auto-detection never overrides a manual pick. */}} {{ $l.name }} {{ else }} {{/* Known language but no translation link for this page: send the reader to that language's home page instead of a dead URL. */}} {{ $l.name }} {{ end }}
    • {{ end }} {{/* planned languages that do not yet exist: shown disabled */}} {{ range $planned }} {{ if not (isset $real .lang) }}
  • {{ .name }} {{ i18n "ComingSoon" | default "Coming soon" }}
    • {{ end }} {{ end }}
